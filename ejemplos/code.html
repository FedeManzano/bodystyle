        $("#repetir, #pass, #nombre, #apellido, #email").keyup( () => {
            
            let clave = $("#pass").val()
            let rep = $("#repetir").val()
            if( 
                $("#nombre").val()    !== ""  &&  $("#nombre").val()    !== undefined   && 
                $("#apellido").val()  !== ""  &&  $("#apellido").val()  !== undefined   &&
                $("#email").val()     !== ""  &&  $("#email").val()     !== undefined   &&
                $("#pass").val()      !== ""  &&  $("#pass").val()      !== undefined   &&
                clave === rep
            )
            {
                $("#enviar").removeClass("desactivado")
            }   
            else 
            {
                $("#enviar").addClass("desactivado")
            }
        })
